Quick Start

Get Started in 30 Seconds

Works with any OpenAI-compatible client. Just change the base URL and API key.

curl https://api.novaai.example.com/v1/chat/completions \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer sk-your-api-key" \
  -d '{
    "model": "gpt-4o",
    "messages": [{"role": "user", "content": "Hello!"}],
    "stream": true
  }'
import openai
client = openai.OpenAI(
    base_url="https://api.novaai.example.com/v1",
    api_key="sk-your-api-key"
)
stream = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "Hello!"}],
    stream=True
)
for chunk in stream:
    print(chunk.choices[0].delta.content, end="")
const OpenAI = require('openai');
const client = new OpenAI({
  baseURL: 'https://api.novaai.example.com/v1',
  apiKey: 'sk-your-api-key'
});
const stream = await client.chat.completions.create({
  model: 'gpt-4o',
  messages: [{ role: 'user', content: 'Hello!' }],
  stream: true,
});
for await (const chunk of stream) {
  process.stdout.write(chunk.choices[0]?.delta?.content || '');
}
